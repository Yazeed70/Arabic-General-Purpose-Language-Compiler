متغير الرصيد = 1000؛
متغير الهدف = 1500؛
متغير الشهر = 1؛
متغير يستطيع_السفر = خطأ؛

اطبع("--- الاختبار الخامس: برنامج إبداعي يحسب التوفير للسفر ---")؛
اطبع("الرصيد الحالي:")؛
اطبع(الرصيد)؛
اطبع("المبلغ المطلوب للسفر:")؛
اطبع(الهدف)؛
اطبع("بدء الادخار (150 كل شهر)...")؛

بينما (الرصيد < الهدف) {
    الرصيد = الرصيد + 150؛
    الشهر = الشهر + 1؛
    
    إذا (الرصيد >= الهدف) {
        يستطيع_السفر = صح؛
    }
}

إذا (يستطيع_السفر == صح) {
    اطبع("مبروك! لقد وصلت للهدف وتقدر تسافر.")؛
    اطبع("عدد الأشهر المستغرقة:")؛
    اطبع(الشهر - 1)؛
    اطبع("الرصيد النهائي أصبح:")؛
    اطبع(الرصيد)؛
} وإلا {
    اطبع("لا يزال عليك التوفير.")؛
}
